        MoveJ Home, v1000, z50, tool0\WObj:=wobj0;

        WaitTime 0.5;

        MoveL pPlacePosFine, v500, z0, tool0\WObj:=wobj0;
        Reset doGripper;
        WaitTime 0.5;
        
        ! 返回 home 位置
        MoveL pPlacePos, v500, z30, tool0\WObj:=wobj0;
        MoveJ Home, v1000, z50, tool0\WObj:=wobj0;
    ENDPROC
ENDMODULE
